Cuando marcás un contacto como Cerrado, se exporta automáticamente a Google Sheets como respaldo histórico.

Seguí estos pasos: abrí Google Sheets → Extensiones → Apps Script → pegá el script de abajo → Implementar → Nueva implementación → Aplicación web → Ejecutar como "Yo" → Acceso "Cualquier persona" → Copiar URL.
Ver script de Google Apps Script ▼ 
function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  
  // Crear encabezados si la hoja está vacía
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['Nombre','Rubro','Email','WhatsApp','Instagram','Canal','Estado','Notas','Fecha cierre']);
  }
  
  sheet.appendRow([
    data.nombre || '',
    data.rubro || '',
    data.email || '',
    data.whatsapp || '',
    data.instagram || '',
    data.canal || '',
    data.estado || '',
    data.notas || '',
    new Date().toLocaleDateString('es-AR')
  ]);
  
  return ContentService
    .createTextOutput(JSON.stringify({ok: true}))
    .setMimeType(ContentService.MimeType.JSON);
}
